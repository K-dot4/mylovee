    let tries = 0;

    function escapeNo() {
        const no = document.getElementById("no");
        tries++;

        const x = Math.random() * 260 - 130;
        const y = Math.random() * 120 - 60;

        no.style.transform = `translate(${x}px, ${y}px)`;

            no.style.display = "none";
        }
    }
